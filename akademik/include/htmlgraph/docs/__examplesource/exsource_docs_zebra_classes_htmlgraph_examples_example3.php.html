  2.  
  3.     // include the HTMLGraph class

  6.     
  7.     
  8.     // watch as the width is specified as percent!

  10.     
  11.     
  12.  
  13.     // now create a BarPlot object (this is the canvas for bars)

  15.     
  16.     // set the colors of all bars added to this plot to red
  17.         $plot->bars->color = "red";
  18.  
  19.     // the easiest way to add some values to a BarPlot object
  20.         $values = array(70, 1112, 1052, 345);
  21.     
  22.     // add some labels too
  23.         $labels = array("p1_1", "p1_2", "p1_3", "p1_4");
  24.  
  25.     // add the values the plot
  26.         $plot->add($values, $labels);
  27.  
  28.     // now create a second BarPlot object
  29.         $plot2 = new HTMLGraph_BarPlot();
  30.  
  31.     // set the colors of all bars added to this plot to green
  32.         $plot2->bars->color = "green";
  33.  
  34.     // add some values to the BarPlot object
  35.         $values2 = array(56, 345, 2391, 678);
  36.     
  37.     // add some labels too
  38.         $labels2 = array("p2_1", "p2_2", "p2_3", "p2_4");
  39.  
  40.     // add the values the plot
  41.         $plot2->add($values2, $labels2);
  42.  
  43.     // create a BarPlotGroup object
  44.         $group = new HTMLGraph_BarPlotGroup();
  45.     
  46.     // add the plots to the group
  47.         $group->add($plot);
  48.     $group->add($plot2);
  49.     
  50.     // add the group to the graph
  51.         $gr->add($group);
  52.  
  53.     // add a footnote to the graph
