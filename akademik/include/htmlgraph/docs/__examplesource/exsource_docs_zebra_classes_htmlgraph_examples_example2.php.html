example

  1. <?php
  2.  
  3.     // include the HTMLGraph class
  4.         require "../class.htmlgraph.php";
  5.     
  6.     // create a new graph of 300 pixels width and 350 pixels in height and with a title
  7.         $gr = new HTMLGraph("300", "350", "Test graph");
  8.     
  9.     // if you want to override the settings from the template related to title
  10.         $gr->title->style = "text-decoration:underline";
  11.     
  12.     // now create a BarPlot object (this is the canvas for bars)
  13.         $plot = new HTMLGraph_BarPlot();
  14.     
  15.     // the easiest way to add some values to a BarPlot object
  16.         $values = array(70, 1112, 1052, 345);
  17.     
  18.     // set some labels for the bars
  19.         $labels = array("label 1", "label 2", "label 3", "label 4");
  20.     
  21.     // also set custom colors for the first, second and last bar
  22.     // the third bar will be drawn with the color set by the template
  23.         $colors = array("red", "green", "", "blue");
  24.  
  25.     // add the values and the colors to the plot
  26.         $plot->add($values, $labels, $colors);
  27.     
  28.     // you can also add a bar using the following method
  29.         $plot->add("500", "a bar added in a different way");
  30.     
  31.     // add the plot to the graph
  32.         $gr->add($plot);
  33.  
  34.     // add a footnote to the graph
  35.         $gr->footnote->caption = "generated by Zebra Framework PHP HTMLGraph Class 1.0 BETA";
  36.     
  37.     // output the graph
  38.         $gr->render();
  39.  
  40. ?>
